ax  ay  az  0
bx  by  bz  0
cx  cy  cz  0
tx  ty  tz  1

p' = p * M

x' = x*ax + y*bx + z*cx + 1*tx
y' = x*ay + y*by + z*cy + 1*ty
z' = x*az + y*bz + z*cz + 1*tz
w' = x*0  + y*0  + z*0  + 1*1
